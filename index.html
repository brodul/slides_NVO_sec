SPI


from pyb import SPI
spi = SPI(1, SPI.MASTER, baudrate=600000, polarity=1, phase=0, crc=0x7)

data = spi.send_recv(b'1234')        # send 4 bytes and receive 4 bytes
buf = bytearray(4)
spi.send_recv(b'1234', buf)          # send 4 bytes and receive 4 into buf
spi.send_recv(buf, buf)              # send/recv 4 bytes from/to buf

import pyb
import esp

# Connect to wifi
esp.connect('brodul_lab', 'lablablab')

def socket_callback(socket, data):
    print(data)

raw_http = """POST / HTTP/1.1
User-Agent: ESP8266 madafakkaa
Host: pizza.brodul.org
Accept: */*
MIME-Version: 1.0
Content-type: text/plain; charset=utf-8

I want pizza.
"""
pin = pyb.Pin(0, pyb.Pin.OUT_PP)
while True:
    if pin.value():
        soc = esp.socket()
        soc.onrecv(socket_callback)
        soc.connect(('192.40.56.62', 8081))
        soc.send(raw_http)
        pyb.delay(1000)
        del soc
    pyb.delay(20)
